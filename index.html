1 Rectangular Mesh Display

üìã What it does:

  • Creates a grid of rectangles on screen
  • Uses nested loops to draw mesh pattern

üí° Key Concepts:

2D Arrays, Nested Loops, GL_LINE_LOOP for drawing rectangles

üì∫ Output Preview

What you'll see:

Expected Output: A white window (500x500) with a blue grid mesh pattern. You'll see 20√ó30 rectangles arranged in rows and columns, creating a mesh-like pattern. Each rectangle is drawn with blue lines (GL_LINE_LOOP). 

// Program 1: Draw a Rectangular Mesh (Grid)
#include <stdio.h>
#include <stdlib.h>
#include <GL/glut.h>

#define maxx 20  // Number of columns
#define maxy 30  // Number of rows
#define dx 10    // Horizontal spacing
#define dy 15    // Vertical spacing

GLfloat x[maxx] = {0.0}, y[maxy] = {0.0};  // Arrays to store coordinates
GLfloat x0 = 50, y0 = 50;  // Starting position
GLint i, j;

// Initialize OpenGL settings
void init() {
    glClearColor(1, 1, 1, 1);  // White background
    glClear(GL_COLOR_BUFFER_BIT);
    glPointSize(5);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0, 499, 0, 499);  // Set 2D view area
}

// Draw the mesh
void display(void) {
    // Calculate x coordinates
    for(i = 0; i < maxx; i++)
        x[i] = x0 + i * dx;
    
    // Calculate y coordinates
    for(j = 0; j < maxy; j++)
        y[j] = y0 + j * dy;
    
    // Draw rectangles using nested loops
    for(i = 0; i < maxx-1; i++)
        for(j = 0; j < maxy-1; j++) {
            glColor3f(0, 0, 1);  // Blue color
            glBegin(GL_LINE_LOOP);  // Start drawing rectangle
                glVertex2f(x[i], y[j]);        // Bottom-left
                glVertex2f(x[i], y[j+1]);      // Top-left
                glVertex2f(x[i+1], y[j+1]);    // Top-right
                glVertex2f(x[i+1], y[j]);      // Bottom-right
            glEnd();
        }
    glFlush();  // Force execution of GL commands
}

int main(int argc, char *argv[]) {
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Rectangular Mesh");
    glutInitWindowPosition(0, 0);
    glutDisplayFunc(display);  // Register display callback
    init();
    glutMainLoop();  // Enter event loop
    return 0;
}
2 Midpoint Line Drawing Algorithm

üìã What it does:

  • Draws a line using Bresenham's midpoint algorithm
  • Efficient - uses only integer arithmetic
  • Takes start and end points as input

üí° Key Concepts:

Decision parameter (d), Incremental algorithm, Pixel plotting

üì∫ Output Preview

What you'll see:

Expected Output: A white window with a black line drawn from your start point to end point.
Example Input: x0=100, y0=100, x1=400, y1=300
The line is drawn pixel by pixel using Bresenham's algorithm - very smooth and efficient! 

// Program 2: Midpoint Line Drawing Algorithm
#include <stdio.h>
#include <GL/glut.h>
#include <stdlib.h>

int x00, x01, y00, y01;  // Line endpoints

// Initialize OpenGL
void init() {
    glClearColor(1, 1, 1, 1);  // White background
    glClear(GL_COLOR_BUFFER_BIT);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(-500, 500, -500, 500);  // Set coordinate system
}

// Plot a single pixel
void writepixel(int x, int y) {
    glPointSize(5);
    glBegin(GL_POINTS);
        glColor3f(0, 0, 0);  // Black color
        glVertex2f(x, y);
    glEnd();
    glFlush();
}

// Draw line using midpoint algorithm
void display() {
    int dx = x01 - x00;  // Change in x
    int dy = y01 - y00;  // Change in y
    int d = 2*dy - dx;   // Initial decision parameter
    int incrE = dy;      // Increment if d < 0
    int incrNE = dy - dx;  // Increment if d >= 0
    int x = x00, y = y00;
    
    writepixel(x, y);  // Plot first point
    
    // Loop until we reach end point
    while(x < x01) {
        if(d < 0) {  // Choose E (East)
            d = d + 2*incrE;
            x++;
        } else {  // Choose NE (North-East)
            d = d + 2*incrNE;
            y++;
            x++;
        }
        writepixel(x, y);  // Plot next point
    }
}

int main(int argc, char *argv[]) {
    // Get user input
    printf("Enter the values \n");
    printf("x0=");
    scanf("%d", &x00);
    printf("y0=");
    scanf("%d", &y00);
    printf("x1=");
    scanf("%d", &x01);
    printf("y1=");
    scanf("%d", &y01);
    
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Midpoint Line");
    glutInitWindowPosition(0, 0);
    glutDisplayFunc(display);
    init();
    glutMainLoop();
    return 0;
}
3 2D Rotation - House Example

üìã What it does:

  • Draws a house shape (base + roof + door)
  • Rotates the house by user-specified angle
  • Shows original (black) and rotated (red) versions

üí° Key Concepts:

2D Transformation, Rotation Matrix, glRotatef(), glTranslatef()

üì∫ Output Preview

What you'll see:

Expected Output: Two houses displayed - original in BLACK and rotated version in RED.
Example Input: Enter rotation angle = 45 degrees
The red house will be rotated around the bottom-left corner by your specified angle. 

// Program 3: 2D House Rotation
#include <stdio.h>
#include <GL/glut.h>

GLfloat theta = 0.0f;  // Rotation angle

// Draw a simple house
void drawHouse() {
    // Draw house base (rectangle)
    glBegin(GL_LINE_LOOP);
        glVertex2f(100.0f, 100.0f);  // Bottom-left
        glVertex2f(250.0f, 100.0f);  // Bottom-right
        glVertex2f(250.0f, 300.0f);  // Top-right
        glVertex2f(100.0f, 300.0f);  // Top-left
    glEnd();

    // Draw roof (triangle)
    glBegin(GL_LINE_LOOP);
        glVertex2f(100.0f, 300.0f);   // Left base
        glVertex2f(175.0f, 400.0f);   // Apex
        glVertex2f(250.0f, 300.0f);   // Right base
    glEnd();

    // Draw door (rectangle)
    glBegin(GL_LINE_LOOP);
        glVertex2f(150.0f, 100.0f);  // Bottom-left
        glVertex2f(150.0f, 150.0f);  // Top-left
        glVertex2f(200.0f, 150.0f);  // Top-right
        glVertex2f(200.0f, 100.0f);  // Bottom-right
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    
    // Draw original house in black
    glColor3f(0.0f, 0.0f, 0.0f);
    drawHouse();

    // Draw rotated house in red
    glPushMatrix();  // Save current state
        glTranslatef(100.0f, 100.0f, 0.0f);   // Move to pivot point
        glRotatef(theta, 0.0f, 0.0f, 1.0f);   // Rotate around Z-axis
        glTranslatef(-100.0f, -100.0f, 0.0f); // Move back
        
        glColor3f(1.0f, 0.0f, 0.0f);  // Red color
        drawHouse();
    glPopMatrix();  // Restore state

    glFlush();
}

void init() {
    glClearColor(1.0f, 1.0f, 1.0f, 1.0f);  // White background
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0.0, 500.0, 0.0, 500.0);
}

int main(int argc, char* argv[]) {
    printf("Enter rotation angle (degrees): ");
    scanf("%f", &theta);
    
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Rotated House");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
4 Scan-Line Polygon Filling

üìã What it does:

  • Fills a polygon using scan-line algorithm
  • Takes 4 vertices as input (quadrilateral)
  • Finds left and right edges for each scan line

üí° Key Concepts:

Scan-line algorithm, Edge detection, Slope inverse, Fill between edges

üì∫ Output Preview

What you'll see:

Expected Output: A filled polygon (quadrilateral) with blue color.
Example Input: Enter 4 vertices like (100,100), (200,100), (220,200), (80,200)
The polygon outline is drawn first, then filled using scan-line algorithm - you'll see it fill line by line! 

// Program 4: Scan-Line Polygon Filling
#include <GL/glut.h>
#include <stdio.h>

int X0, Y0, X1, Y1, X2, Y2, X3, Y3;  // 4 vertices of polygon

void init() {
    glutInitWindowSize(500, 500);
    glutCreateWindow("Scan-Line Polygon Filling Algorithm");
    glClearColor(1, 1, 1, 1);  // White background
    glClear(GL_COLOR_BUFFER_BIT);
    gluOrtho2D(0, 500, 0, 500);
    glColor3f(0, 0, 1);  // Blue color
    glPointSize(1);
}

// Detect edges and store left/right boundaries
void edgeDetect(int x1, int y1, int x2, int y2, int LE[], int RE[]) {
    double slopeInverse, x;  // 1/m (inverse slope)
    int i, temp;
    
    // Ensure y1 < y2 (swap if needed)
    if(y2 < y1) {
        temp = y1; y1 = y2; y2 = temp;
        temp = x1; x1 = x2; x2 = temp;
    }
    
    // Calculate inverse slope (dx/dy)
    if(y1 != y2)
        slopeInverse = (x2 - x1) / (double)(y2 - y1);
    else
        slopeInverse = x2 - x1;
    
    x = x1;
    // For each scan line, update left and right edges
    for(i = y1; i <= y2; i++) {
        if(x < LE[i])  LE[i] = x;  // Update left edge
        if(x > RE[i])  RE[i] = x;  // Update right edge
        x += slopeInverse;  // Move to next scan line
    }
}

// Draw a single pixel
void drawPixel(int x, int y) {
    glBegin(GL_POINTS);
        glVertex2f(x, y);
    glEnd();
}

// Fill polygon using scan-line algorithm
void scanLineFill() {
    int i, j, LE[500], RE[500];
    
    // Initialize left and right edge arrays
    for(i = 0; i < 500; i++) {
        LE[i] = 500;  // Left edge starts at right
        RE[i] = 0;    // Right edge starts at left
    }
    
    // Process all 4 edges of polygon
    edgeDetect(X0, Y0, X1, Y1, LE, RE);
    edgeDetect(X1, Y1, X2, Y2, LE, RE);
    edgeDetect(X2, Y2, X3, Y3, LE, RE);
    edgeDetect(X3, Y3, X0, Y0, LE, RE);
    
    // Fill between left and right edges
    for(i = 0; i < 500; i++)
        if(LE[i] <= RE[i])
            for(j = LE[i]; j <= RE[i]; j++) {
                drawPixel(j, i);
                glFlush();
            }
}

void display() {
    // Draw polygon outline
    glBegin(GL_LINE_LOOP);
        glVertex2f(X0, Y0);
        glVertex2f(X1, Y1);
        glVertex2f(X2, Y2);
        glVertex2f(X3, Y3);
    glEnd();
    glFlush();
    scanLineFill();  // Fill the polygon
}

int main(int argc, char *argv[]) {
    printf("Enter co-ordinates of polygon vertices (0 <= input < 500) :\n");
    printf("X0="); scanf("%d", &X0);
    printf("Y0="); scanf("%d", &Y0);
    printf("X1="); scanf("%d", &X1);
    printf("Y1="); scanf("%d", &Y1);
    printf("X2="); scanf("%d", &X2);
    printf("Y2="); scanf("%d", &Y2);
    printf("X3="); scanf("%d", &X3);
    printf("Y3="); scanf("%d", &Y3);
    
    glutInit(&argc, argv);
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}

  • Press 'P' ‚Üí Draw Pentagon

üí° Key Concepts:

Event handling, glutMouseFunc(), glutKeyboardFunc(), Interactive graphics

üì∫ Output Preview

What you'll see:

Expected Output: Interactive drawing canvas!
üñ±Ô∏è Left Click: Red points | Right Click: Blue lines
‚å®Ô∏è Press T: Magenta triangle | Press S: Cyan square | Press P: Yellow pentagon

// Program 5: Mouse and Keyboard Interaction
#include <GL/glut.h>
#include <stdio.h>

void init() {
    glutInitWindowSize(500, 500);
    glutCreateWindow("Mouse and Keyboard Operations");
    glClearColor(1, 1, 1, 1);  // White background
    glClear(GL_COLOR_BUFFER_BIT);
    gluOrtho2D(0, 500, 500, 0);  // Note: Y-axis is inverted
}

// Draw a point at (x, y)
void drawPoint(int x, int y) {
    glColor3f(1, 0, 0);  // Red
    glPointSize(10);
    glBegin(GL_POINTS);
        glVertex2f(x, y);
    glEnd();
    glFlush();
}

// Draw a horizontal line at (x, y)
void drawLine(int x, int y) {
    glColor3f(0, 0, 1);  // Blue
    glLineWidth(10);
    glBegin(GL_LINES);
        glVertex2f(x - 25, y);  // Start point
        glVertex2f(x + 25, y);  // End point
    glEnd();
    glFlush();
}

// Draw triangle centered at (x, y)
void drawTriangle(int x, int y) {
    glColor3f(1, 0, 1);  // Magenta
    glBegin(GL_TRIANGLES);
        glVertex2f(x, y - 10);        // Top vertex
        glVertex2f(x - 10, y + 10);   // Bottom-left
        glVertex2f(x + 10, y + 10);   // Bottom-right
    glEnd();
    glFlush();
}

// Draw square centered at (x, y)
void drawSquare(int x, int y) {
    glColor3f(0, 1, 1);  // Cyan
    glBegin(GL_POLYGON);
        glVertex2f(x - 10, y - 10);  // Bottom-left
        glVertex2f(x - 10, y + 10);  // Top-left
        glVertex2f(x + 10, y + 10);  // Top-right
        glVertex2f(x + 10, y - 10);  // Bottom-right
    glEnd();
    glFlush();
}

// Draw pentagon centered at (x, y)
void drawPentagon(int x, int y) {
    glColor3f(1, 1, 0);  // Yellow
    glBegin(GL_POLYGON);
        glVertex2f(x, y - 15);
        glVertex2f(x - 10, y - 5);
        glVertex2f(x - 10, y + 10);
        glVertex2f(x + 10, y + 10);
        glVertex2f(x + 10, y - 5);
    glEnd();
    glFlush();
}

// Mouse event handler
void mouse(int button, int state, int x, int y) {
    if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN)
        drawPoint(x, y);
    if (button == GLUT_RIGHT_BUTTON && state == GLUT_DOWN)
        drawLine(x, y);
}

// Keyboard event handler
void keyboard(unsigned char key, int x, int y) {
    if (key == 't' || key == 'T')
        drawTriangle(x, y);
    if (key == 's' || key == 'S')
        drawSquare(x, y);
    if (key == 'p' || key == 'P')
        drawPentagon(x, y);
}

void display() {
    glFlush();
}

int main(int argc, char *argv[]) {
    printf("Mouse Functions :\n\tLEFT CLICK - Point\n\tRIGHT CLICK - Line\n");
    printf("Keyboard Functions :\n\tT or t - Triangle\n\tS or s - Square\n\tP or p - Pentagon\n");
    
    glutInit(&argc, argv);
    init();
    glutDisplayFunc(display);
    glutMouseFunc(mouse);      // Register mouse handler
    glutKeyboardFunc(keyboard);  // Register keyboard handler
    glutMainLoop();
    return 0;
}
6 Midpoint Circle Drawing Algorithm

üìã What it does:

  • Draws a circle using midpoint algorithm
  • Uses 8-way symmetry (draws 8 points at once)
  • Only integer arithmetic - very efficient!

üí° Key Concepts:

Circle symmetry, Decision parameter, Plotting 8 symmetric points

üì∫ Output Preview

What you'll see:

Expected Output: A perfect RED circle on BLACK background.
Example Input: Enter radius = 100
The circle is drawn using 8-way symmetry - very efficient! Only 1/8 of the circle is calculated, the rest is mirrored. Larger radius = bigger circle! 

// Program 6: Midpoint Circle Drawing
#include <stdio.h>
#include <stdlib.h>
#include <GL/glut.h>

int r;  // Radius of circle

void init() {
    glClearColor(0, 0, 0, 0);  // Black background
    glClear(GL_COLOR_BUFFER_BIT);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(-900, 900, -900, 900);
}

// Plot a single pixel
void writepixel(int x, int y) {
    glPointSize(5);
    glBegin(GL_POINTS);
        glColor3f(1, 0, 0);  // Red color
        glVertex2f(x, y);
    glEnd();
    glFlush();
}

// Draw circle using midpoint algorithm
void midpointcircle() {
    int x = 0, y = r;
    double d = 5.0/4.0 - r;  // Initial decision parameter
    
    while(y > x) {
        // Update decision parameter
        if(d < 0)
            d += 2*x + 3.0;  // Choose E (East)
        else {
            d += 2*(x - y) + 5.0;  // Choose SE (South-East)
            y--;
        }
        x++;
        
        // Plot 8 symmetric points (circle symmetry)
        writepixel(x, y);      // Octant 1
        writepixel(y, x);      // Octant 2
        writepixel(-x, y);     // Octant 3
        writepixel(y, -x);     // Octant 4
        writepixel(-x, -y);    // Octant 5
        writepixel(-y, -x);    // Octant 6
        writepixel(x, -y);     // Octant 7
        writepixel(-y, x);     // Octant 8
    }
}

int main(int argc, char* argv[]) {
    printf("Enter radius: ");
    scanf("%d", &r);
    
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Circle");
    init();
    glutDisplayFunc(midpointcircle);
    glutMainLoop();
    return 0;
}
7 Sierpinski Gasket (Fractal Triangle)

üìã What it does:

  • Creates a fractal pattern using recursion
  • Divides triangle into 3 smaller triangles
  • Repeats the process n times
  • Higher n = more detailed pattern

üí° Key Concepts:

Recursion, Fractals, Midpoint calculation, Divide and conquer

üì∫ Output Preview

What you'll see:

Expected Output: A beautiful fractal triangle pattern (Sierpinski Gasket).
Example Input: n = 3 (try values 0-5)
Higher n = more subdivisions = more detailed fractal pattern!
‚Ä¢ n=0: Simple triangle | n=3: Complex pattern | n=5: Very detailed (but slow) 

// Program 7: Sierpinski Gasket (Recursive Fractal)
#include <stdio.h>
#include <stdlib.h>
#include <GL/glut.h>

int n;  // Number of recursive subdivisions

void init() {
    glClearColor(1, 1, 1, 1);  // White background
    glClear(GL_COLOR_BUFFER_BIT);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0, 10, 0, 10);
}

// Draw a single triangle
void triangle(float *a, float *b, float *c) {
    glVertex2f(a[0], a[1]);  // Vertex A
    glVertex2f(b[0], b[1]);  // Vertex B
    glVertex2f(c[0], c[1]);  // Vertex C
}

// Recursive function to subdivide triangle
void draw_triangle(float *a, float *b, float *c, int k) {
    float ab[2], ac[2], bc[2];  // Midpoints
    
    if(k > 0) {  // If more subdivisions needed
        // Calculate midpoint of AB
        for(int i = 0; i < 2; i++)
            ab[i] = (a[i] + b[i]) / 2;
        
        // Calculate midpoint of AC
        for(int i = 0; i < 2; i++)
            ac[i] = (a[i] + c[i]) / 2;
        
        // Calculate midpoint of BC
        for(int i = 0; i < 2; i++)
            bc[i] = (c[i] + b[i]) / 2;

        // Recursively draw 3 smaller triangles
        draw_triangle(a, ab, ac, k-1);   // Top triangle
        draw_triangle(b, bc, ab, k-1);   // Bottom-left
        draw_triangle(c, bc, ac, k-1);   // Bottom-right
    } else {
        triangle(a, b, c);  // Base case: draw the triangle
    }
}

void display() {
    float a[2] = {1, 1};      // Bottom-left vertex
    float b[2] = {6, 1};      // Bottom-right vertex
    float c[2] = {3.5, 5};    // Top vertex

    glBegin(GL_TRIANGLES);
        glColor3f(0, 0, 0);  // Black color
        draw_triangle(a, b, c, n);
    glEnd();
    glFlush();
}

int main(int argc, char* argv[]) {
    printf("Enter number of subdivisions (0-5 recommended): ");
    scanf("%d", &n);
    
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Sierpinski Gasket");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
